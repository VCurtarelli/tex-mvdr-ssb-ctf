\begin{algorithm}[H]
\caption{RCKP beamforming algorithm}\label{alg:RCKP_beamformer}
\begin{algorithmic} 
\Require \\%
    $\bvh{1} ~,~ \bvh{2} ~,~ \cdots ~,~ \bvh{K}$ \Comment{Input beamformers} \\%
    $\My{1} ~,~ \My{2} ~,~ \cdots ~,~ \My{K}$ \Comment{Beamformer $\y$-axis sizes}
\Ensure \\%
    $\bvh$ \Comment{Full-array beamformer} \\%
    $\Mx ~,~ \My$ \Comment{Output beamformer sizes}
\Procedure%
    \State $\bvH \att [[1]]$ \Comment{$\sz{1}{1}$ matrix}
    \State $\My \att 1$
    \For{$0 \leq k < K$}
        \State $\bvH' \att \ivect{\My{k}}{\bvh{k}}$
        \State $\bvH \att \bvH \bconv \bvH'$
        \State $\My \att \My + \My{k} - 1$
    \EndFor
    \State $\bvh \att \vect{\bvH}$
    \State $M \att \len{\bvh}$ \Comment{Length of output vector}
    \State $\Mx \att \frac{M}{\My}$
\end{algorithmic}
\end{algorithm}
