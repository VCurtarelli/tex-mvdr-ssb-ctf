\let\mrm\symup
\def\mtimes{\mathbin{\scalebox{0.7}{$\times$}}}
\NewDocumentCommand{\sz}{s m m}{%
	\IfBooleanTF{#1}{%
		\bts{{#2} \mtimes {#3}}%
	}{%
		{#2} \mtimes {#3}%
	}%
}

\NewDocumentCommand{\el}{m g O{m} o}{%
	\bts{#1}_{%
		\IfValueTF{#4}{%
			(#3\IfValueT{#2}{_{#2}},#4\IfValueT{#2}{_{#2}})%
		}{%
			#3\IfValueT{#2}{_{#2}}%
		}        
	}
}
