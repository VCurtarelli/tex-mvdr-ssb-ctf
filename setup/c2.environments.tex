\NewDocumentEnvironment{subalign}{g}{%
	\subequations%
		\IfValueT{#1}{\label{#1}}%
		\allowdisplaybreaks%
		\align%
	}{%
		\endalign%
	\endsubequations%
}%
\NewDocumentEnvironment{subgather}{g}{%
	\subequations%
		\IfValueT{#1}{\label{#1}}%
		\allowdisplaybreaks%
		\gather%
	}{%
		\endgather%
	\endsubequations%
}%
\NewDocumentEnvironment{equations}{g}{%
	\equation%
		\IfValueT{#1}{\label{#1}}%
		\allowdisplaybreaks%
		\aligned%
	}{%
		\endaligned%
	\endequation%
}
